 31 #define DEFAULT_PASSES  3
 32 #define DEV_RANDOM      "/dev/urandom"
 33 #define MAX_CHUNK       4096
 34

 36 static int         wipe(char *);

 43 /*
 44  * overwrite a file with one or more passes of random data, then unlink it.
 45  */
 46 int
 47 main(int argc, char **argv)
 48 {
 49     size_t passes, tmp_passes, wiped, failed, i;
 50     char **file_ptr, **wipe_success, **wipe_fail;
 51     int retval, opt, verbose;
 52 
 53     passes  = DEFAULT_PASSES;
 54     retval  = EX_DATAERR;
 55     wiped   = 0;
 56     failed  = 0;
 57     verbose = 0;
 58 
 59     if (argc == 1)
 60         usage();
 61 
 62     while (-1 != (opt = getopt(argc, argv, "hn:vV"))) {
 63         switch( opt ) {
 64         case 'h':
 65             usage();
 66             break;              /* don't technically need it but meh */
 67         case 'n':
 68             tmp_passes = atoi(optarg);
 69             passes = tmp_passes > 0 ? tmp_passes : passes;
 70             break;
 71         case 'v':
 72             verbose = 1;
 73             break;
 74         case 'V':
 75             version();
 76             exit(EX_OK);
 77         default:
 78             usage();
 79             /* NOTREACHED */
 80         }
 81     }
 82 
 83     argc -= optind;
 84     argv += optind;
 85 
 86     file_ptr = argv;
 87     wipe_success = calloc(argc, sizeof(char *));
 88     wipe_fail    = calloc(argc, sizeof(char *));
 89 
 90 
 91     while (NULL != *file_ptr) {
 92         printf("%s: ", *file_ptr);
 93         fflush(stdout);
 94 
 95         if (EX_OK != do_wipe(*file_ptr, passes)) {
 96             wipe_fail[failed] = strdup(*file_ptr);
 97             failed++;
 98         } else {
 99             wipe_success[wiped] = strdup(*file_ptr);
100             wiped++;
101             printf(" OK!");
102         }
103         file_ptr++;
104         printf("\n");
105     }
106 
107     if (verbose) {
108         if (0 < wiped) {
109             printf("success: ");
110             for( i = 0; i < wiped; ++i ) {
111                 printf("%s ", wipe_success[i]);
112             }
113             printf("\n");
114         }
115 
116         if (0 < failed) {
117             printf("failures: ");
118             for( i = 0; i < failed; ++i ) {
119                 printf("%s ", wipe_fail[i]);
120             }
121             printf("\n");
122         }
123     }
124 
125     /* free allocated memory */
126     for (i = 0; i < failed; ++i) {
127         free(wipe_fail[i]);
128         wipe_fail[i] = NULL;
129     }
130     free(wipe_fail);
131     wipe_fail = NULL;
132 
133     for (i = 0; i < wiped; ++i) {
134         free(wipe_success[i]);
135         wipe_success[i] = NULL;
136     }
137     free(wipe_success);
138     wipe_success = NULL;
139 
140     return retval;
141 }

143 /*
144  * takes a filename and the number of passes to wipe the file
145  */
146 int
147 do_wipe(char *filename, size_t passes)
148 {
149     struct stat sb;
150     size_t filesize, i;
151     int retval;
152 
153     retval = EX_IOERR;
154     if (-1 == stat(filename, &sb)) {
155         warn("%s", filename);
156         return retval;
157     }
158     filesize = sb.st_size;
159 
160     for (i = 0; i < passes; ++i) {
161         if (EX_OK != wipe(filename)) {
162             printf("!");
163             return retval;
164         } else if (-1 == stat(filename, &sb)) {
165             printf("!");
166             return retval;
167         } else if (filesize != (size_t)sb.st_size) {
168             printf("!");
169             return retval;
170         }
171         printf(".");
172         fflush(stdout);
173     }
174 
175     if (-1 == truncate(filename, (off_t)0))
176         warn("%s", filename);
177 
178     if (-1 == unlink(filename)) {
179         warn("%s", filename);
180     } else {
181         retval = EX_OK;
182     }
183 
184     return retval;
185 }

187 /*
188  * takes a filename and attempts to overwrite it with random data
189  */
190 int
191 wipe(char *filename)
192 {
193     struct stat sb;
194 
195     size_t chunk, filesize, rdsz, wiped, wrsz;
196     FILE *devrandom, *target;
197     int retval, targetfd;
198     char *rdata;
199 
200     retval = EX_IOERR;
201     wiped  = 0;
202 
203     if (-1 == stat(filename, &sb))
204         return retval;
205 
206     filesize = sb.st_size;
207 
208     /* 
209      * open devrandom first: if this fails, we don't want to touch the target
210      * file. 
211      */
212     devrandom = fopen(DEV_RANDOM, "r");
213     if (NULL == devrandom || -1 == ferror(devrandom)) {
214         warn("failed to open PRNG %s!", DEV_RANDOM);
215         return retval;
216     }
217 
218     /*
219      * for security purposes, we want to make sure to actually overwrite the
220      * the file. r+ gives us read/write but more importantly, sets the write
221      * stream at the beginning of the file. a side note is that when overwriting
222      * a file, its size will never seem to change.
223      */
224     target   = fopen(filename, "r+");
225     if (NULL == target || -1 == ferror(target)) {
226         warn("failed to open %s", filename);
227         fclose(devrandom);
228         return retval;
229     }
230 
231     rewind(target);
232     targetfd = fileno(target);
233 
234     /* wait to calloc until we really need the data - makes cleaning up less
235      * tricky.
236      */
237     rdata = calloc(MAX_CHUNK, sizeof(char));
238     if (NULL == rdata) {
239         warn("could not allocate random data memory");
240         fclose(devrandom);
241         fclose(target);
242         return retval;
243     }
244 
245     while (wiped < filesize) {
246         chunk = filesize - wiped;
247         chunk = chunk > MAX_CHUNK ? MAX_CHUNK : chunk;
248 
249         rdsz  =  fread( rdata, sizeof(char), chunk, devrandom );
250         wrsz  = fwrite( rdata, sizeof(char), chunk, target );
251 
252         if (-1 == stat(filename, &sb)) {
253             warn(" stat on %s failed", filename);
254             break;
255         }
256         if ((rdsz != wrsz) || (filesize != (unsigned int)sb.st_size)) {
257             warn("invalid read/write size");
258             break;
259         }
260 
261         wiped += chunk;
262     }
263 
264     if ((0 != fclose(devrandom)) || (0 != fclose(target)))
265         warn("%s", filename);
266     else
267         retval = EX_OK;
268 
269     free(rdata);
270     rdata = NULL;
271 
272     return retval;
273 }

275 /*
276  * print a quick usage message 
277  */
278 void
279 usage()
280 {
281     version();
282     printf("usage: %s [-v] [-n number] files\n", __progname);
283     printf("        -n passes   specify number of passes\n");
284     printf("                    (default is %d passes)\n", DEFAULT_PASSES);
285     printf("        -v          display list of failures and wiped files"
286            " after wiping\n"
287            "                    (verbose mode).\n");
288     printf("        -V          print version information.");
289 
290     printf("\n");
291     exit(EX_USAGE);
292 }
